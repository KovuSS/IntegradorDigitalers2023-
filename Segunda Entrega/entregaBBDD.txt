SELECT CLIENTES.NOMBRE, CLIENTES.APELLIDO, BICICLETAS.MARCA, BICICLETAS.MODELO, TIPO_DE_BICICLETA.DESCRIPCION
FROM CLIENTES 
INNER JOIN BICICLETAS
ON CLIENTES.CODIGO_DE_BICI = BICICLETAS.ID
INNER JOIN TIPO_DE_BICICLETA
ON BICICLETAS.TIPO = TIPO_DE_BICICLETA.CODIGO;
